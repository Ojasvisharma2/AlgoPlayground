#include <iostream>
#include <vector>
#include <algorithm>
#include <climits>
using namespace std;

// 1. Kadane's Algorithm - Maximum Subarray Sum
int maxSubarraySum(const vector<int>& arr) {
    int maxSoFar = INT_MIN;
    int currSum = 0;

    for (int x : arr) {
        currSum += x;
        maxSoFar = max(maxSoFar, currSum);
        if (currSum < 0) currSum = 0;
    }
    return maxSoFar;
}

// 2. Two Pointers - Pair Sum in Sorted Array
bool pairSumSorted(const vector<int>& arr, int target, pair<int, int>& result) {
    int left = 0, right = arr.size() - 1;

    while (left < right) {
        int sum = arr[left] + arr[right];
        if (sum == target) {
            result = {arr[left], arr[right]};
            return true;
        } else if (sum < target) {
            left++;
        } else {
            right--;
        }
    }
    return false;
}

// 3. Sliding Window - Maximum sum of subarray of size k
int maxSumSubarrayOfSizeK(const vector<int>& arr, int k) {
    int n = arr.size();
    if (k > n || k <= 0) return -1;

    int windowSum = 0;
    for (int i = 0; i < k; i++) {
        windowSum += arr[i];
    }

    int maxSum = windowSum;
    for (int i = k; i < n; i++) {
        windowSum += arr[i] - arr[i - k];
        maxSum = max(maxSum, windowSum);
    }
    return maxSum;
}

// 4. Binary Search
int binarySearch(const vector<int>& arr, int target) {
    int left = 0, right = arr.size() - 1;

    while (left <= right) {
        int mid = left + (right - left) / 2;
        if (arr[mid] == target) return mid;
        else if (arr[mid] < target) left = mid + 1;
        else right = mid - 1;
    }
    return -1;
}

// 5. Greedy - Minimum Coins
vector<int> minCoinsGreedy(int amount, vector<int> coins) {
    sort(coins.rbegin(), coins.rend());
    vector<int> used;

    for (int coin : coins) {
        while (amount >= coin) {
            amount -= coin;
            used.push_back(coin);
        }
    }

    if (amount != 0) return {};
    return used;
}

// 6. DP - Climbing Stairs
int climbingStairs(int n) {
    if (n <= 2) return n;

    int prev2 = 1, prev1 = 2, curr;

    for (int i = 3; i <= n; i++) {
        curr = prev1 + prev2;
        prev2 = prev1;
        prev1 = curr;
    }
    return prev1;
}

// 7. Recursion - Subset Sum
bool subsetSumRec(vector<int>& arr, int index, int target) {
    if (target == 0) return true;
    if (index == arr.size() || target < 0) return false;

    return subsetSumRec(arr, index + 1, target - arr[index]) ||
           subsetSumRec(arr, index + 1, target);
}

    cout << "Enter elements: ";

    return arr;
}

                    cout << "Found at index " << index << "\n";
